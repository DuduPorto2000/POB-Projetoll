CREATE TABLE MENSAGEM (ID  SERIAL NOT NULL, DATAHORA VARCHAR(255), TEXTO VARCHAR(255), CRIADOR_NOMESENHA VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE ADMINISTRADOR (NOMESENHA VARCHAR(255) NOT NULL, ATIVO BOOLEAN, EMAIL VARCHAR(255), PRIMARY KEY (NOMESENHA))
CREATE TABLE LOG (ID  SERIAL NOT NULL, DATAHORA VARCHAR(255), NOME VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE USUARIO (NOMESENHA VARCHAR(255) NOT NULL, ATIVO BOOLEAN, PRIMARY KEY (NOMESENHA))
