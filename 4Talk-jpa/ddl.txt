CREATE TABLE MENSAGEM (ID  SERIAL NOT NULL, DATAHORA VARCHAR(255), TEXTO VARCHAR(255), CRIADOR_ID INTEGER, PRIMARY KEY (ID))
CREATE TABLE ADMINISTRADOR (ID  SERIAL NOT NULL, ATIVO BOOLEAN, EMAIL VARCHAR(255), NOMESENHA VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE LOG (ID  SERIAL NOT NULL, DATAHORA VARCHAR(255), NOME VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE USUARIO (ID  SERIAL NOT NULL, ATIVO BOOLEAN, NOMESENHA VARCHAR(255), PRIMARY KEY (ID))
